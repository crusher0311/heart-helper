(async function deleteNewJobs() {
  console.log('üóëÔ∏è Starting cleanup of "New Job" entries...');
  
  let deletedCount = 0;
  let errorCount = 0;
  
  while (true) {
    // Find the checkbox label with text "New Job"
    const newJobLabels = Array.from(document.querySelectorAll('label')).filter(label => {
      const text = label.textContent?.trim() || '';
      return text === 'New Job';
    });
    
    if (newJobLabels.length === 0) {
      console.log(`‚úÖ Cleanup complete! Deleted ${deletedCount} jobs, ${errorCount} errors`);
      break;
    }
    
    console.log(`Found ${newJobLabels.length} "New Job" entries - deleting one (${deletedCount + 1})...`);
    
    // Get the first New Job label
    const newJobLabel = newJobLabels[0];
    
    // Find the parent row/container
    let jobRow = newJobLabel;
    for (let i = 0; i < 15; i++) {
      jobRow = jobRow.parentElement;
      if (!jobRow) break;
      
      // Look for the trash/delete button (it's usually an SVG icon or button)
      const trashButtons = Array.from(jobRow.querySelectorAll('button, [role="button"]')).filter(btn => {
        // Check if button contains trash icon or has delete-related attributes
        const svg = btn.querySelector('svg');
        const hasTrashIcon = svg && (
          svg.innerHTML.includes('trash') || 
          svg.innerHTML.includes('delete') ||
          svg.querySelector('path')
        );
        
        return hasTrashIcon || 
               btn.getAttribute('aria-label')?.toLowerCase().includes('delete') ||
               btn.title?.toLowerCase().includes('delete');
      });
      
      if (trashButtons.length > 0) {
        // Usually the last button on the right is the trash
        const trashButton = trashButtons[trashButtons.length - 1];
        console.log('  ‚úì Found trash button, clicking...');
        trashButton.click();
        await new Promise(r => setTimeout(r, 800));
        
        // Look for the red DELETE button in modal
        const deleteButtons = Array.from(document.querySelectorAll('button')).filter(btn => {
          const text = btn.textContent.trim().toUpperCase();
          const style = window.getComputedStyle(btn);
          const isRed = style.backgroundColor.includes('rgb(239') || // Red button
                       style.backgroundColor.includes('rgb(220') ||
                       btn.className.includes('danger') ||
                       btn.className.includes('destructive');
          
          return text === 'DELETE' && isRed;
        });
        
        if (deleteButtons.length > 0) {
          console.log('  ‚úì Clicking DELETE button...');
          deleteButtons[0].click();
          deletedCount++;
          await new Promise(r => setTimeout(r, 1200)); // Wait for deletion
          break;
        } else {
          console.log('  ‚ö†Ô∏è No DELETE button found in modal');
          // Try to close modal
          const cancelBtn = Array.from(document.querySelectorAll('button')).find(b => 
            b.textContent.trim() === 'CANCEL'
          );
          if (cancelBtn) cancelBtn.click();
          errorCount++;
          await new Promise(r => setTimeout(r, 500));
          break;
        }
      }
    }
    
    // If still no trash button found
    if (!jobRow || errorCount > 0) {
      console.log('  ‚ö†Ô∏è Skipping this job...');
      newJobLabel.style.opacity = '0.3'; // Mark as attempted
      await new Promise(r => setTimeout(r, 300));
      errorCount = 0; // Reset for next attempt
    }
    
    // Progress update
    if (deletedCount % 5 === 0 && deletedCount > 0) {
      console.log(`üìä Progress: ${deletedCount} deleted`);
    }
    
    // Safety break
    if (deletedCount + errorCount > 500) {
      console.log('‚ö†Ô∏è Reached safety limit (500), stopping...');
      break;
    }
  }
})();
